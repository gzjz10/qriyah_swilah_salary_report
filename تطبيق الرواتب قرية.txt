function init(){
  const DEPTS=[...new Set(DATA.map(e=>e.dept))];
  const BRANCHES=[...new Set(DATA.map(e=>e.branch))];
  const total = DATA.reduce((s,e)=>s+e.salary,0);

  document.getElementById('headerStats').innerHTML=`
    <div class="hstat"><div class="hstat-val">${DATA.length}</div><div class="hstat-lbl">إجمالي الموظفين</div></div>
    <div class="hstat"><div class="hstat-val">${fmt(total)}</div><div class="hstat-lbl">إجمالي الرواتب</div></div>
    <div class="hstat"><div class="hstat-val">${DEPTS.length}</div><div class="hstat-lbl">الأقسام</div></div>
    <div class="hstat"><div class="hstat-val">${BRANCHES.length}</div><div class="hstat-lbl">الفروع</div></div>
  `;

  // Mini stats
  const deptStats={};
  DATA.forEach(e=>{if(!deptStats[e.dept])deptStats[e.dept]={count:0,total:0};deptStats[e.dept].count++;deptStats[e.dept].total+=e.salary;});
  const clsList=['gold','green','blue','purple','orange','teal','pink','gold','blue','green','purple','orange'];
  document.getElementById('miniStats').innerHTML=Object.entries(deptStats).map(([d,s],i)=>`
    <div class="stat-card ${clsList[i%clsList.length]}" onclick="filterByDept('${esc(d)}')">
      <div class="stat-ico">${getDeptIcon(d)}</div>
      <div class="stat-val">${s.count}</div>
      <div class="stat-lbl">${d}</div>
      <div class="stat-sub">${fmt(s.total)}</div>
    </div>`).join('');

  // Populate filters
  ['deptFilter','cardDeptFilter'].forEach(id=>{
    const sel=document.getElementById(id);
    sel.innerHTML='<option value="">كل الأقسام</option>';
    DEPTS.forEach(d=>{const o=document.createElement('option');o.value=d;o.textContent=d;sel.appendChild(o);});
  });
  ['branchFilter','cardBranchFilter'].forEach(id=>{
    const sel=document.getElementById(id);
    sel.innerHTML='<option value="">كل الفروع</option>';
    BRANCHES.forEach(b=>{const o=document.createElement('option');o.value=b;o.textContent=b;sel.appendChild(o);});
  });

  filterTable();
  filterCards();
  buildDeptPanel();
  buildAnalytics();
  buildReports();
}

// ── TABLE ──
function filterTable(){
  const q=document.getElementById('searchInput').value.trim().toLowerCase();
  const dp=document.getElementById('deptFilter').value;
  const br=document.getElementById('branchFilter').value;
  let rows=DATA.filter(e=>{
    const mQ=!q||(e.name||'').toLowerCase().includes(q)||(e.title||'').toLowerCase().includes(q);
    const mD=!dp||e.dept===dp;
    const mB=!br||e.branch===br;
    return mQ&&mD&&mB;
  });
  if(sortDesc)rows.sort((a,b)=>b.salary-a.salary);
  else rows.sort((a,b)=>a.salary-b.salary);
  const maxSal=Math.max(...DATA.map(e=>e.salary));
  document.getElementById('tableCount').textContent=`${rows.length} موظف | إجمالي: ${fmt(rows.reduce((s,e)=>s+e.salary,0))}`;
  const allSorted=[...DATA].sort((a,b)=>b.salary-a.salary);
  const tb=document.getElementById('empTable');
  if(!rows.length){tb.innerHTML=`<tr><td colspan="7"><div class="empty-state"><div class="emo">🔍</div><p>لا توجد نتائج</p></div></td></tr>`;return;}
  tb.innerHTML=rows.map(e=>{
    const rank=allSorted.findIndex(x=>x.id===e.id)+1;
    const rc=rank===1?'rank-1':rank===2?'rank-2':rank===3?'rank-3':'rank-other';
    const pct=Math.round(e.salary/maxSal*100);
    return `<tr onclick="openEditModal(${e.id})">
      <td><div class="rank-badge ${rc}">${rank<=3?['🥇','🥈','🥉'][rank-1]:rank}</div></td>
      <td><div class="emp-name">${e.name}</div></td>
      <td><div class="emp-role">${e.title}</div></td>
      <td><span class="dept-tag ${deptCls(e.dept)}" style="border:1px solid;border-radius:8px;font-size:11px;font-weight:700;padding:3px 8px">${getDeptIcon(e.dept)} ${e.dept}</span></td>
      <td style="font-size:12px;color:var(--text2)">${BRANCH_ICONS[e.branch]||'🏢'} ${e.branch}</td>
      <td><div class="salary-cell">${fmt(e.salary)}<div class="salary-bar"><div class="salary-fill" style="width:${pct}%"></div></div></div></td>
      <td style="display:flex;gap:6px;padding:10px 18px">
        <button class="btn-raise" style="padding:5px 10px;font-size:11px" onclick="event.stopPropagation();openEditModal(${e.id})">✏️</button>
        <button class="btn-delete" style="padding:5px 10px;font-size:11px" onclick="event.stopPropagation();confirmDelete(${e.id})">🗑️</button>
      </td>
    </tr>`;
  }).join('');
}

function toggleSort(){
  sortDesc=!sortDesc;
  document.getElementById('sortBtn').textContent=sortDesc?'↑ من الأعلى راتبًا':'↓ من الأقل راتبًا';
  filterTable();
}

// ── CARDS ──
function getInitials(name){return(name||'').split(' ').slice(0,2).map(w=>w[0]||'').join('');}

function filterCards(){
  const q=(document.getElementById('cardSearch').value||'').trim().toLowerCase();
  const dp=document.getElementById('cardDeptFilter').value;
  const br=document.getElementById('cardBranchFilter').value;
  const sort=document.getElementById('cardSortFilter').value;
  let rows=DATA.filter(e=>{
    const mQ=!q||(e.name||'').toLowerCase().includes(q)||(e.title||'').toLowerCase().includes(q);
    const mD=!dp||e.dept===dp;
    const mB=!br||e.branch===br;
    return mQ&&mD&&mB;
  });
  if(sort==='sal-desc')rows.sort((a,b)=>b.salary-a.salary);
  else if(sort==='sal-asc')rows.sort((a,b)=>a.salary-b.salary);
  else rows.sort((a,b)=>a.name.localeCompare(b.name,'ar'));
  document.getElementById('cardsCount').textContent=`${rows.length} بطاقة`;
  const allSorted=[...DATA].sort((a,b)=>b.salary-a.salary);
  const grid=document.getElementById('cardsGrid');
  if(!rows.length){grid.innerHTML=`<div class="empty-state" style="grid-column:1/-1"><div class="emo">🔍</div><p>لا توجد نتائج</p></div>`;return;}
  grid.innerHTML=rows.map(e=>{
    const rank=allSorted.findIndex(x=>x.id===e.id)+1;
    const rc=rank===1?'rank-1':rank===2?'rank-2':rank===3?'rank-3':'rank-other';
    const col=DEPT_COLORS[e.dept]||'#E8B86D';
    return `<div class="emp-card">
      ${rank<=3?`<div class="emp-rank-badge ${rc}">${['🥇','🥈','🥉'][rank-1]}</div>`:''}
      <div class="emp-card-header">
        <div class="emp-avatar" style="background:${col}22;color:${col};border-color:${col}44">${getInitials(e.name)}</div>
        <div>
          <div class="emp-card-name">${e.name}</div>
          <div class="emp-card-role">${e.title}</div>
        </div>
      </div>
      <div class="emp-card-body">
        <div class="emp-card-row">
          <span class="ecr-label">القسم</span>
          <span class="dept-tag ${deptCls(e.dept)}" style="border:1px solid;border-radius:6px;font-size:10px;font-weight:700;padding:2px 8px">${getDeptIcon(e.dept)} ${e.dept}</span>

        <div class="emp-card-row">
          <span class="ecr-label">الفرع</span>
          <span class="ecr-value">${BRANCH_ICONS[e.branch]||'🏢'} ${e.branch}</span>

// ── ANALYTICS ──
function buildAnalytics(){
  const deptTotals={};
  DATA.forEach(e=>{deptTotals[e.dept]=(deptTotals[e.dept]||0)+e.salary;});
  const deptEntries=Object.entries(deptTotals).sort((a,b)=>b[1]-a[1]);
  const total=deptEntries.reduce((s,[,v])=>s+v,0);
  const COLS=Object.values(DEPT_COLORS);
  let offset=0;const r=70,cx=80,cy=80;
  let paths='',legend='';
  deptEntries.forEach(([d,val],i)=>{
    const pct=val/total;const dash=pct*2*Math.PI*r;const gap=(1-pct)*2*Math.PI*r;
    const col=DEPT_COLORS[d]||COLS[i%COLS.length];
    paths+=`<circle cx="${cx}" cy="${cy}" r="${r}" fill="none" stroke="${col}" stroke-width="18" stroke-dasharray="${dash.toFixed(1)} ${gap.toFixed(1)}" stroke-dashoffset="${(-offset*2*Math.PI*r).toFixed(1)}" style="cursor:pointer" onclick="openDeptModal('${esc(d)}')"/>`;
    offset+=pct;
    legend+=`<div class="legend-item" onclick="openDeptModal('${esc(d)}')">
      <div class="legend-dot" style="background:${col}"></div>
      <span class="legend-lbl">${d}</span>
      <span class="legend-val" style="color:${col}">${fmt(val)}</span>
    </div>`;
  });
  document.getElementById('deptDonut').innerHTML=`
    <svg viewBox="0 0 160 160" width="160" height="160">
      <circle cx="80" cy="80" r="70" fill="none" stroke="#1A2236" stroke-width="18"/>
      ${paths}
      <text x="80" y="76" text-anchor="middle" fill="#E8B86D" font-size="11" font-weight="800" font-family="Cairo">${deptEntries.length}</text>
      <text x="80" y="91" text-anchor="middle" fill="#6B7A96" font-size="9" font-family="Cairo">أقسام</text>
    </svg>
    <div class="donut-legend">${legend}</div>`;

  // Branch bars
  const brTotals={};
  DATA.forEach(e=>{brTotals[e.branch]=(brTotals[e.branch]||0)+e.salary;});
  const brEntries=Object.entries(brTotals).sort((a,b)=>b[1]-a[1]);
  const maxBR=brEntries[0]?.[1]||1;
  document.getElementById('branchBars').innerHTML=brEntries.map(([br,val])=>`
    <div class="bar-row">
      <div class="bar-lbl">${BRANCH_ICONS[br]||'🏢'} ${br}</div>
      <div class="bar-track"><div class="bar-fill" style="width:${(val/maxBR*100).toFixed(0)}%;background:#60A5FA"></div></div>
      <div class="bar-val">${fmt(val)}</div>
    </div>`).join('');

  // Top 10
  const top10=[...DATA].sort((a,b)=>b.salary-a.salary).slice(0,10);
  const maxTop=top10[0]?.salary||1;
  document.getElementById('top10Bars').innerHTML=top10.map((e,i)=>`
    <div class="bar-row">
      <div class="bar-lbl" title="${e.name}">${e.name.split(' ')[0]}</div>
      <div class="bar-track" onclick="openEditModal(${e.id})"><div class="bar-fill" style="width:${(e.salary/maxTop*100).toFixed(0)}%;background:${i<3?'#E8B86D':'#3B82F6'}"></div></div>
      <div class="bar-val">${fmt(e.salary)}</div>
    </div>`).join('');

  // Ranges
  const ranges=[{l:'أكثر من 10,000',min:10000,max:Infinity},{l:'5,000 - 10,000',min:5000,max:10000},{l:'3,000 - 5,000',min:3000,max:5000},{l:'2,000 - 3,000',min:2000,max:3000},{l:'أقل من 2,000',min:0,max:2000}];
  const rc2=['#E8B86D','#22C55E','#60A5FA','#A78BFA','#6B7A96'];
  const rCounts=ranges.map(r=>({...r,count:DATA.filter(e=>e.salary>=r.min&&e.salary<r.max).length}));
  const maxR=Math.max(...rCounts.map(r=>r.count));
  document.getElementById('rangeBars').innerHTML=rCounts.map((r,i)=>`
    <div class="bar-row">
      <div class="bar-lbl">${r.l}</div>
      <div class="bar-track"><div class="bar-fill" style="width:${(r.count/maxR*100).toFixed(0)}%;background:${rc2[i]}"></div></div>
      <div class="bar-val">${r.count} موظف</div>
    </div>`).join('');
}

function buildPrintReport(){
  const allStats={
    count:DATA.length,
    total:DATA.reduce((s,e)=>s+e.salary,0),
    max:Math.max(...DATA.map(e=>e.salary)),
    avg:Math.round(DATA.reduce((s,e)=>s+e.salary,0)/DATA.length)
  };

  const groups={};
  DATA.forEach(e=>{
    groups[e.dept]=groups[e.dept]||[];
    groups[e.dept].push(e);
  });
  Object.keys(groups).forEach(d=>groups[d].sort((a,b)=>b.salary-a.salary));

  const fmtN = n => n.toLocaleString('ar-SA');

  let summaryCards='';
  Object.entries(groups).forEach(([dept,emps])=>{
    const tot=emps.reduce((s,e)=>s+e.salary,0);
    summaryCards+=`
      <div class="pr-summary-card">
        <h4>${getDeptIcon(dept)} ${dept}</h4>
        <div class="pr-sc-stats">
          <div class="pr-sc-stat"><div class="lbl">عدد الموظفين</div><div class="val">${emps.length}</div></div>
          <div class="pr-sc-stat"><div class="lbl">إجمالي الرواتب</div><div class="val">${fmtN(tot)}</div></div>
        </div>
      </div>`;
  });

  let deptSections='';
  Object.entries(groups).forEach(([dept,emps])=>{
    const tot=emps.reduce((s,e)=>s+e.salary,0);
    const avg=Math.round(tot/emps.length);
    const max=Math.max(...emps.map(e=>e.salary));
    // Split large depts
    const chunkSize=25;
    for(let i=0;i<emps.length;i+=chunkSize){
      const chunk=emps.slice(i,i+chunkSize);
      const isFirst=i===0;
      deptSections+=`
        <div class="pr-dept-section" style="${!isFirst?'page-break-before:always':''}">
          <div class="pr-dept-header">📁 قسم ${dept}${emps.length>chunkSize?` (جزء ${Math.floor(i/chunkSize)+1})`:''}
          </div>
          <div class="pr-dept-content">
            ${isFirst?`
            <div class="pr-dept-stats-grid">
              <div class="pr-dept-stat"><div class="lbl">👥 عدد الموظفين</div><div class="val">${emps.length}</div></div>
              <div class="pr-dept-stat"><div class="lbl">💰 إجمالي الرواتب</div><div class="val">${fmtN(tot)}</div><div style="color:#7f8c8d;font-size:0.8rem">ريال</div></div>
              <div class="pr-dept-stat"><div class="lbl">📈 متوسط الراتب</div><div class="val">${fmtN(avg)}</div><div style="color:#7f8c8d;font-size:0.8rem">ريال</div></div>
              <div class="pr-dept-stat"><div class="lbl">🏆 أعلى راتب</div><div class="val">${fmtN(max)}</div><div style="color:#7f8c8d;font-size:0.8rem">ريال</div></div>
            </div>`:''}
            <div class="pr-table-wrap">
              <table class="pr-table">
                <thead><tr><th>#</th><th>المسمى الوظيفي</th><th>الفرع</th><th>اسم الموظف</th><th>الراتب (ريال)</th></tr></thead>
                <tbody>
                  ${chunk.map((emp,j)=>`
                    <tr>
                      <td><span class="pr-rank-badge">${i+j+1}</span></td>
                      <td>${emp.title}</td>
                      <td><span class="pr-branch-tag">${emp.branch}</span></td>
                      <td>${emp.name}</td>
                      <td><span class="pr-salary-badge">${fmtN(emp.salary)}</span></td>
                    </tr>`).join('')}
                </tbody>
              </table>
            </div>
          </div>
        </div>`;
    }
  });

  let overallCards='';
  Object.entries(groups).forEach(([dept,emps])=>{
    const tot=emps.reduce((s,e)=>s+e.salary,0);
    overallCards+=`
      <div class="pr-overall-card">
        <h4>${getDeptIcon(dept)} ${dept}</h4>
        <div class="stats-grid">
          <div class="stat-box"><div class="lbl">عدد الموظفين</div><div class="val">${emps.length}</div></div>
          <div class="stat-box"><div class="lbl">إجمالي الرواتب</div><div class="val">${fmtN(tot)} ريال</div></div>
        </div>
      </div>`;
  });

  document.getElementById('print-report').innerHTML=`
    <div style="max-width:100%;direction:rtl;font-family:'Tajawal',sans-serif">
      <div class="pr-main-header">
        <h1>📊 تقرير المرتبات والأجور حسب الأقسام - شركة قرية صويلح</h1>
        <div class="pr-meta">إجمالي الموظفين: ${allStats.count} | إجمالي الرواتب: ${fmtN(allStats.total)} ريال | تاريخ التقرير: ${new Date().toLocaleDateString('ar-SA')}</div>
      </div>

      <div class="pr-global-stats">
        <div class="pr-stat-card"><h3>📈 متوسط الراتب العام</h3><div class="pr-val">${fmtN(allStats.avg)}</div><div style="color:#7f8c8d;font-size:0.85rem">ريال سعودي</div></div>
        <div class="pr-stat-card"><h3>🏢 عدد الأقسام</h3><div class="pr-val">${Object.keys(groups).length}</div><div style="color:#7f8c8d;font-size:0.85rem">قسم وإدارة</div></div>
        <div class="pr-stat-card"><h3>🏆 أعلى راتب في الشركة</h3><div class="pr-val">${fmtN(allStats.max)}</div><div style="color:#7f8c8d;font-size:0.85rem">ريال سعودي</div></div>
        <div class="pr-stat-card"><h3>💰 إجمالي الرواتب</h3><div class="pr-val">${fmtN(allStats.total)}</div><div style="color:#7f8c8d;font-size:0.85rem">ريال سعودي</div></div>
      </div>

      <div class="pr-section-title">📋 ملخص الأقسام والإدارات</div>
      <div class="pr-summary-cards">${summaryCards}</div>

      ${deptSections}

      <div style="page-break-before:always">
        <div class="pr-overall-summary">
          <h2>📊 الملخص النهائي لجميع الأقسام</h2>
          <div class="pr-overall-cards">${overallCards}</div>
        </div>
      </div>
    </div>`;
}

  const curDept=document.getElementById('deptFilter').value;
  const curBranch=document.getElementById('branchFilter').value;
  ['deptFilter','cardDeptFilter'].forEach(id=>{
    const sel=document.getElementById(id);
    const cur=sel.value;
    sel.innerHTML='<option value="">كل الأقسام</option>';
    DEPTS.forEach(d=>{const o=document.createElement('option');o.value=d;o.textContent=d;if(d===cur)o.selected=true;sel.appendChild(o);});
  });
  ['branchFilter','cardBranchFilter'].forEach(id=>{
    const sel=document.getElementById(id);
    const cur=sel.value;
    sel.innerHTML='<option value="">كل الفروع</option>';
    BRANCHES.forEach(b=>{const o=document.createElement('option');o.value=b;o.textContent=b;if(b===cur)o.selected=true;sel.appendChild(o);});
  });

  filterTable();
  filterCards();

  const total=DATA.reduce((s,e)=>s+e.salary,0);
  document.getElementById('headerStats').innerHTML=`

    <div class="hstat"><div class="hstat-val">${fmt(total)}</div><div class="hstat-lbl">إجمالي الرواتب</div></div>
    <div class="hstat"><div class="hstat-val">${DEPTS.length}</div><div class="hstat-lbl">الأقسام</div></div>
    <div class="hstat"><div class="hstat-val">${BRANCHES.length}</div><div class="hstat-lbl">الفروع</div></div>
  `;
  // Rebuild mini stats
  const deptStats={};
  DATA.forEach(e=>{if(!deptStats[e.dept])deptStats[e.dept]={count:0,total:0};deptStats[e.dept].count++;deptStats[e.dept].total+=e.salary;});

      <div class="stat-lbl">${d}</div>
      <div class="stat-sub">${fmt(s.total)}</div>
    </div>`).join('');
}
